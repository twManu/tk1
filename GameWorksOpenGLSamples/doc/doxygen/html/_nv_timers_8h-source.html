00091     void getResults() {
00092         // Make a pass over all timers - if any are pending results ("in flight"), then
00093         // grab the the time diff and add to the accumulator.  Then, mark the timer as 
00094         // not in flight and record it as a possible next timer to use (since it is
00095         // now unused)
00096         int32_t freeQuery = -1;
00097         for (unsigned int i = 0; i < TIMER_COUNT; i++) {
00098             // Are we awaiting a result?
00099             if (m_timeStampQueryInFlight[i]) {
00100                 GLuint available = false;
00101                 m_glGetQueryObjectuiv(m_queries[i][TIMESTAMP_QUERY_END], NV_QUERY_RESULT_AVAILABLE, &available);
00102 
00103                 // Is the result ready?
00104                 if (available) {
00105                     uint64_t timeStart, timeEnd;
00106                     m_glGetQueryObjectui64v(m_queries[i][TIMESTAMP_QUERY_BEGIN], NV_QUERY_RESULT, &timeStart);
00107                     m_glGetQueryObjectui64v(m_queries[i][TIMESTAMP_QUERY_END], NV_QUERY_RESULT, &timeEnd);
00108                 
00109                     // add the time to the accumulator
00110                     m_elapsedCycles += float(double(timeEnd - timeStart) * 1.e-6);
00111                     freeQuery = i;
00112                     m_timeStampQueryInFlight[i] = false;
00113 
00114                     // Increment the counter of start/stop cycles
00115                     m_startStopCycles++;
00116                 }
00117             }
00118         }
00119         // Use the newly-freed counters if one exists
00120         // otherwise, swap to the "next" counter
00121         if (freeQuery >= 0)
00122         {
00123             m_nextTimeStampQuery = freeQuery;
00124         }
00125         else
00126         {
00127             m_nextTimeStampQuery++;
00128             m_nextTimeStampQuery = (m_nextTimeStampQuery >= TIMER_COUNT) ? 0 : m_nextTimeStampQuery;
00129         }
00130     }
